Usage Instructions

1. Include the CSS

<link rel="stylesheet" href="css/acaistack-components.css">

2. Include the JavaScript

<script src="js/acaistack-components.js"></script>

3. Listen to Custom Events (Optional)

// Agent cards emit events you can listen to
document.addEventListener('agent-demo-request', (e) => {
  console.log('Demo requested for:', e.detail.agentName);
});

document.addEventListener('agent-purchase-request', (e) => {
  console.log('Purchase requested:', e.detail);
});

// Pricing cards
document.addEventListener('pricing-select', (e) => {
  console.log('Plan selected:', e.detail.tierName);
});

// Re-initialize after dynamic content load
window.AcaiStack.init();
